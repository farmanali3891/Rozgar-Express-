const express = require("express"); const cors = require("cors"); const fetch = require("node-fetch"); const app = express(); app.use(cors()); const API_KEY = "PASTE_YOUR_RAPIDAPI_KEY"; app.get("/jobs", async (req, res) => { try { const url = "https://jsearch.p.rapidapi.com/search?query=jobs%20Pakistan%20Dubai%20Saudi&num_pages=1"; const response = await fetch(url, { method: "GET", headers: { "X-RapidAPI-Key": API_KEY, "X-RapidAPI-Host": "jsearch.p.rapidapi.com", }, }); const data = await response.json(); res.json(data); } catch (err) { res.status(500).json({ error: "Failed to fetch jobs" }); } }); app.listen(3000, () => console.log("Server running on port 3000"));
